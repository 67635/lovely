yesButton.addEventListener('click', function onYesClick() { text.textContent = 'Ты ж моя умничка!'; text.style.marginTop = '20px'; gif.src = 'love1.gif'; animateGif(gif); yesButton.remove(); noButton.remove(); yesButton.removeEventListener('click', onYesClick); // Получаем границы текстового элемента const textRect = text.getBoundingClientRect(); // Функция для создания и анимации гифок function createDancingGif() { const dancingGif = document.createElement('img'); dancingGif.src = 'kot1.gif'; // Убедитесь, что гифка 'kot1.gif' доступна dancingGif.style.position = 'absolute'; dancingGif.style.width = '100px'; // Устанавливаем размер гифки dancingGif.style.height = 'auto'; // Генерация случайных координат, исключая область текста let x, y; do { x = Math.random() * (window.innerWidth - dancingGif.width); y = Math.random() * (window.innerHeight - dancingGif.height); } while ( x + dancingGif.width > textRect.left && x < textRect.right && y + dancingGif.height > textRect.top && y < textRect.bottom ); dancingGif.style.left = `${x}px`; dancingGif.style.top = `${y}px`; document.body.appendChild(dancingGif); // Анимация движения гифки let xDirection = Math.random() > 0.5 ? 1 : -1; let yDirection = Math.random() > 0.5 ? 1 : -1; const moveGif = () => { let newX = parseFloat(dancingGif.style.left) + xDirection * 1; let newY = parseFloat(dancingGif.style.top) + yDirection * 1; // Проверка, чтобы гифка не заходила на текст if ( newX + dancingGif.width > textRect.left && newX < textRect.right && newY + dancingGif.height > textRect.top && newY < textRect.bottom ) { xDirection *= -1; yDirection *= -1; newX = parseFloat(dancingGif.style.left) + xDirection * 1; newY = parseFloat(dancingGif.style.top) + yDirection * 1; } // Проверка границ экрана if (newX < 0 || newX > window.innerWidth - dancingGif.width) xDirection *= -1; if (newY < 0 || newY > window.innerHeight - dancingGif.height) yDirection *= -1; dancingGif.style.left = `${newX}px`; dancingGif.style.top = `${newY}px`; requestAnimationFrame(moveGif); }; moveGif(); // Удаление гифки через некоторое время setTimeout(() => { dancingGif.remove(); }, 10000); // Гифка исчезает через 10 секунд } // Создаем несколько гифок for (let i = 0; i < 5; i++) { // Создаем 5 гифок setTimeout(createDancingGif, i * 2000); // Каждая новая гифка появляется через 2 секунды } });
